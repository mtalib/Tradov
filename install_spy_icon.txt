#!/bin/bash
# ===============================================================================
# SPYDER - Quick SPY Icon Installation Script
# This script automatically sets up your SPY icon to launch Spyder
# ===============================================================================

set -e

# Configuration
SPYDER_HOME="/home/adam/Projects/Spyder"
SCRIPTS_DIR="$SPYDER_HOME/SpyderQ_Scripts"
ICON_FILE="$SPYDER_HOME/assets/spyder-icon.png"

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
CYAN='\033[0;36m'
NC='\033[0m'

clear

echo -e "${BLUE}╔════════════════════════════════════════════╗${NC}"
echo -e "${BLUE}║      SPYDER SPY ICON INSTALLER v1.0       ║${NC}"
echo -e "${BLUE}╚════════════════════════════════════════════╝${NC}"
echo ""

# Check if icon exists
if [ ! -f "$ICON_FILE" ]; then
    echo -e "${RED}[✗]${NC} Icon not found at: $ICON_FILE"
    echo -e "${YELLOW}[!]${NC} Please ensure spyder-icon.png exists in $SPYDER_HOME/assets/"
    exit 1
else
    echo -e "${GREEN}[✓]${NC} Found SPY icon: spyder-icon.png"
fi

# Check if startup script exists
if [ ! -f "$SCRIPTS_DIR/SpyderQ10_StartAll.sh" ]; then
    echo -e "${RED}[✗]${NC} Startup script not found"
    echo -e "${YELLOW}[!]${NC} Please save SpyderQ10_StartAll.sh first"
    exit 1
else
    echo -e "${GREEN}[✓]${NC} Found startup script"
fi

echo ""
echo -e "${CYAN}Installing SPY icon launcher...${NC}"
echo ""

# Step 1: Make scripts executable
echo -e "${YELLOW}[1/6]${NC} Setting permissions..."
chmod +x "$SCRIPTS_DIR/SpyderQ10_StartAll.sh" 2>/dev/null || {
    echo -e "${RED}[✗]${NC} Failed to set permissions"
    exit 1
}
echo -e "${GREEN}[✓]${NC} Permissions set"

# Step 2: Create desktop launcher file
echo -e "${YELLOW}[2/6]${NC} Creating desktop launcher..."
cat > "$SPYDER_HOME/spyder-trading.desktop" << EOF
[Desktop Entry]
Version=1.0
Type=Application
Name=Spyder Trading System
Comment=Automated SPY Options Trading Platform
GenericName=SPY Options Trading
Exec=$SCRIPTS_DIR/SpyderQ10_StartAll.sh
Terminal=false
StartupNotify=true
Icon=$ICON_FILE
Categories=Finance;Trading;Application;
Keywords=trading;options;spy;spyder;stocks;market;
Actions=Stop;Status;Monitor;Config;Logs;Terminal;

[Desktop Action Stop]
Name=Stop Trading System
Exec=$SCRIPTS_DIR/SpyderQ11_StopAll.sh
Icon=process-stop

[Desktop Action Status]
Name=Check System Status
Exec=gnome-terminal -- $SCRIPTS_DIR/SpyderQ20_Status.sh
Icon=dialog-information

[Desktop Action Monitor]
Name=Open System Monitor
Exec=gnome-terminal -- $SCRIPTS_DIR/SpyderQ21_Monitor.sh
Icon=utilities-system-monitor

[Desktop Action Config]
Name=Edit Configuration
Exec=gedit $SPYDER_HOME/.env
Icon=preferences-system

[Desktop Action Logs]
Name=View Live Logs
Exec=gnome-terminal -- tail -f $SPYDER_HOME/logs/system/main.log
Icon=text-x-log

[Desktop Action Terminal]
Name=Open Spyder Terminal
Exec=gnome-terminal --working-directory=$SPYDER_HOME -- bash -c "source spyder_venv/bin/activate; exec bash"
Icon=utilities-terminal

StartupWMClass=spyder-trading
X-GNOME-Autostart-enabled=false
EOF

echo -e "${GREEN}[✓]${NC} Desktop launcher created"

# Step 3: Install to user applications
echo -e "${YELLOW}[3/6]${NC} Installing to applications menu..."
mkdir -p ~/.local/share/applications
cp "$SPYDER_HOME/spyder-trading.desktop" ~/.local/share/applications/
chmod +x ~/.local/share/applications/spyder-trading.desktop
echo -e "${GREEN}[✓]${NC} Installed to applications"

# Step 4: Update desktop database
echo -e "${YELLOW}[4/6]${NC} Updating desktop database..."
update-desktop-database ~/.local/share/applications/ 2>/dev/null || true
echo -e "${GREEN}[✓]${NC} Database updated"

# Step 5: Create desktop shortcut (optional)
echo -e "${YELLOW}[5/6]${NC} Creating desktop shortcut..."
DESKTOP_DIR="$HOME/Desktop"
if [ -d "$DESKTOP_DIR" ]; then
    cp "$SPYDER_HOME/spyder-trading.desktop" "$DESKTOP_DIR/"
    chmod +x "$DESKTOP_DIR/spyder-trading.desktop"
    # Trust the desktop file (for GNOME)
    gio set "$DESKTOP_DIR/spyder-trading.desktop" metadata::trusted true 2>/dev/null || true
    echo -e "${GREEN}[✓]${NC} Desktop shortcut created"
else
    echo -e "${YELLOW}[!]${NC} Desktop folder not found, skipping shortcut"
fi

# Step 6: Create command-line shortcut
echo -e "${YELLOW}[6/6]${NC} Creating command shortcut..."
mkdir -p ~/.local/bin
ln -sf "$SCRIPTS_DIR/SpyderQ10_StartAll.sh" ~/.local/bin/spyder 2>/dev/null || true
echo -e "${GREEN}[✓]${NC} Command shortcut created"

# Add to favorites/dock (GNOME specific)
if command -v gsettings &> /dev/null; then
    echo ""
    echo -e "${CYAN}Adding to favorites...${NC}"
    
    # Get current favorites
    CURRENT_FAVS=$(gsettings get org.gnome.shell favorite-apps)
    
    # Check if already in favorites
    if [[ ! "$CURRENT_FAVS" == *"spyder-trading.desktop"* ]]; then
        # Add to favorites
        NEW_FAVS="${CURRENT_FAVS%]}, 'spyder-trading.desktop']"
        gsettings set org.gnome.shell favorite-apps "$NEW_FAVS" 2>/dev/null && {
            echo -e "${GREEN}[✓]${NC} Added to dock/favorites"
        } || {
            echo -e "${YELLOW}[!]${NC} Could not add to favorites automatically"
        }
    else
        echo -e "${GREEN}[✓]${NC} Already in favorites"
    fi
fi

echo ""
echo -e "${GREEN}╔════════════════════════════════════════════╗${NC}"
echo -e "${GREEN}║        INSTALLATION COMPLETE! 🚀          ║${NC}"
echo -e "${GREEN}╚════════════════════════════════════════════╝${NC}"
echo ""
echo "Your SPY icon is now configured!"
echo ""
echo -e "${CYAN}How to launch Spyder:${NC}"
echo "  1. ${GREEN}Click the SPY icon${NC} in your dashboard/dock"
echo "  2. ${GREEN}Search 'Spyder'${NC} in your applications menu"
echo "  3. ${GREEN}Double-click${NC} the desktop shortcut"
echo "  4. ${GREEN}Type 'spyder'${NC} in terminal"
echo ""
echo -e "${CYAN}Right-click the icon for quick actions:${NC}"
echo "  • Stop Trading System"
echo "  • Check System Status"
echo "  • Open System Monitor"
echo "  • Edit Configuration"
echo "  • View Live Logs"
echo "  • Open Spyder Terminal"
echo ""

# Test launch option
read -p "Would you like to test launch Spyder now? (y/N): " -n 1 -r
echo ""
if [[ $REPLY =~ ^[Yy]$ ]]; then
    echo -e "${CYAN}Launching Spyder Trading System...${NC}"
    "$SCRIPTS_DIR/SpyderQ10_StartAll.sh"
else
    echo -e "${YELLOW}To launch manually, click your SPY icon or run:${NC}"
    echo "  $SCRIPTS_DIR/SpyderQ10_StartAll.sh"
fi

echo ""
echo -e "${YELLOW}Note:${NC} If the icon doesn't appear immediately:"
echo "  • Press Alt+F2, type 'r', press Enter (GNOME)"
echo "  • Or log out and log back in"
echo ""
