#!/bin/bash
# ===============================================================================
# SPYDER - SPY Icon Setup Script
# This script configures your SPY dashboard icon to launch Spyder
# ===============================================================================

set -e

# Configuration
SPYDER_HOME="/home/adam/Projects/Spyder"
DESKTOP_FILE="$SPYDER_HOME/spyder-trading.desktop"
SCRIPTS_DIR="$SPYDER_HOME/SpyderQ_Scripts"

# Colors
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

echo -e "${BLUE}================================================${NC}"
echo -e "${BLUE}    SPY ICON SETUP FOR SPYDER${NC}"
echo -e "${BLUE}================================================${NC}"
echo ""

# Step 1: Make startup script executable
echo -e "${YELLOW}[1/5]${NC} Making startup script executable..."
chmod +x "$SCRIPTS_DIR/SpyderQ10_StartAll.sh"
echo -e "${GREEN}✓${NC} Done"

# Step 2: Create desktop launcher
echo -e "${YELLOW}[2/5]${NC} Installing desktop launcher..."
cp "$DESKTOP_FILE" ~/.local/share/applications/
chmod +x ~/.local/share/applications/spyder-trading.desktop
echo -e "${GREEN}✓${NC} Done"

# Step 3: Update desktop database
echo -e "${YELLOW}[3/5]${NC} Updating desktop database..."
update-desktop-database ~/.local/share/applications/ 2>/dev/null || true
echo -e "${GREEN}✓${NC} Done"

# Step 4: Create SPY icon if it doesn't exist
echo -e "${YELLOW}[4/5]${NC} Setting up icon..."
ICON_DIR="$SPYDER_HOME/assets"
ICON_FILE="$ICON_DIR/spy-icon.png"

if [ ! -f "$ICON_FILE" ]; then
    mkdir -p "$ICON_DIR"
    
    # Create a simple SPY icon using ImageMagick if available
    if command -v convert &> /dev/null; then
        convert -size 256x256 xc:navy \
            -fill white -gravity center \
            -pointsize 120 -annotate +0+0 'SPY' \
            -bordercolor white -border 10 \
            "$ICON_FILE"
        echo -e "${GREEN}✓${NC} Created SPY icon"
    else
        echo -e "${YELLOW}!${NC} Icon not found. Add your SPY icon to: $ICON_FILE"
    fi
else
    echo -e "${GREEN}✓${NC} Icon found"
fi

# Step 5: Create symbolic link for easier access
echo -e "${YELLOW}[5/5]${NC} Creating command shortcut..."
sudo ln -sf "$SCRIPTS_DIR/SpyderQ10_StartAll.sh" /usr/local/bin/spyder-start 2>/dev/null || {
    ln -sf "$SCRIPTS_DIR/SpyderQ10_StartAll.sh" ~/.local/bin/spyder-start 2>/dev/null || true
}
echo -e "${GREEN}✓${NC} Done"

# Final instructions
echo ""
echo -e "${GREEN}================================================${NC}"
echo -e "${GREEN}    SETUP COMPLETE!${NC}"
echo -e "${GREEN}================================================${NC}"
echo ""
echo "Your SPY icon should now be functional!"
echo ""
echo "Options to launch Spyder:"
echo "  1. Click the SPY icon in your dashboard"
echo "  2. Search for 'Spyder Trading System' in applications"
echo "  3. Run: spyder-start"
echo "  4. Run: $SCRIPTS_DIR/SpyderQ10_StartAll.sh"
echo ""
echo "Right-click the icon for quick actions:"
echo "  • Stop Trading System"
echo "  • Check System Status"
echo "  • Open System Monitor"
echo "  • Edit Configuration"
echo "  • View Logs"
echo ""

# Test if we can find the desktop file
if [ -f ~/.local/share/applications/spyder-trading.desktop ]; then
    echo -e "${GREEN}✓${NC} Desktop launcher verified"
else
    echo -e "${YELLOW}!${NC} Desktop launcher may need manual installation"
fi

# Refresh desktop icons (for GNOME)
if command -v gnome-shell &> /dev/null; then
    echo "Refreshing GNOME Shell..."
    killall -HUP gnome-shell 2>/dev/null || true
fi

echo ""
echo "If the icon doesn't appear immediately:"
echo "  1. Log out and log back in"
echo "  2. Or press Alt+F2, type 'r', and press Enter (GNOME)"
echo ""
